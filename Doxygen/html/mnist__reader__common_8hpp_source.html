24 inline uint32_t read_header(const std::unique_ptr<char[]>& buffer, size_t position) {
25  auto header = reinterpret_cast<uint32_t*>(buffer.get());
26 
27  auto value = *(header + position);
28  return (value << 24) | ((value << 8) & 0x00FF0000) | ((value >> 8) & 0X0000FF00) | (value >> 24);
29 }

36 inline std::unique_ptr<char[]> read_mnist_file(const std::string& path, uint32_t key) {
37  std::ifstream file;
38  file.open(path, std::ios::in | std::ios::binary | std::ios::ate);
39 
40  if (!file) {
41  std::cout << "Error opening file" << std::endl;
42  return {};
43  }
44 
45  auto size = file.tellg();
46  std::unique_ptr<char[]> buffer(new char[size]);
47 
48  //Read the entire file at once
49  file.seekg(0, std::ios::beg);
50  file.read(buffer.get(), size);
51  file.close();
52 
53  auto magic = read_header(buffer, 0);
54 
55  if (magic != key) {
56  std::cout << "Invalid magic number, probably not a MNIST file" << std::endl;
57  return {};
58  }
59 
60  auto count = read_header(buffer, 1);
61 
62  if (magic == 0x803) {
63  auto rows = read_header(buffer, 2);
64  auto columns = read_header(buffer, 3);
65 
66  if (size < count * rows * columns + 16) {
67  std::cout << "The file is not large enough to hold all the data, probably corrupted" << std::endl;
68  return {};
69  }
70  } else if (magic == 0x801) {
71  if (size < count + 8) {
72  std::cout << "The file is not large enough to hold all the data, probably corrupted" << std::endl;
73  return {};
74  }
75  }
76 
77  return buffer;
78 }
