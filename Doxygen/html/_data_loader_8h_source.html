40  void load_data(Matrix& train_data, Matrix& train_label, Matrix& test_data, Matrix& test_label)
41  {
42  // TODO: Изменить алгоритм подгрузки данных - сделать его более эффективным.
43  // Текущая загрузка - 10.48 сек.
44  mnist::MNIST_dataset<std::vector, std::vector<ImageFormat>, LabelFormat> dataset =
45  mnist::read_dataset<std::vector, std::vector, ImageFormat, LabelFormat>(MNIST_DATA_LOCATION);
46 
47  const long train_size_one_object = dataset.training_images[0].size();
48  const long test_size_one_object = dataset.test_images[0].size();
49  const long train_size = dataset.training_images.size();
50  const long test_size = dataset.test_images.size();
51 
52  // resize matrix
53  train_data.resize(train_size, train_size_one_object);
54  train_label.resize(1, train_size);
55  test_data.resize(test_size, test_size_one_object);
56  test_label.resize(1, test_size);
57 
58  // TRAIN_DATA processing
59  for (int i = 0; i < train_size; i++)
60  {
61  for (int j = 0; j < train_size_one_object; j++)
62  {
63  train_data(i, j) = static_cast<Scalar>(dataset.training_images[i][j]);
64  }
65  }
66 
67  // TRAIN_LABEL processing
68  for (int i = 0; i < train_size; i++)
69  {
70  train_label(0, i) = static_cast<Scalar>(dataset.training_labels[i]);
71  }
72 
73  // TEST_DATA processing
74  for (int i = 0; i < test_size; i++)
75  {
76  for (int j = 0; j < test_size_one_object; j++)
77  {
78  test_data(i, j) = static_cast<Scalar>(dataset.test_images[i][j]);
79  }
80  }
81 
82  // TEST_LABEL processing
83  for (int i = 0; i < test_size; i++)
84  {
85  test_label(0, i) = static_cast<Scalar>(dataset.test_labels[i]);
86  }
87 
88  train_data.transposeInPlace();

90  }
