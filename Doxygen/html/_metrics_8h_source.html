88  Matrix static build_confusion_matrix(const Scalar* real, const Scalar* predict, const long& sample_size)
89  {
90  int tp = 0;
91  int tn = 0;
92  int fp = 0;
93  int fn = 0;
94 
95  for (int i = 0; i < sample_size; i++)
96  {
97  if ((real[i] == 0) && (predict[i] == 0)) tn++;
98  else if ((real[i] == 1) && (predict[i] == 1)) tp++;
99  else if ((real[i] == 0) && (predict[i] == 1)) fp++;
100  else if ((real[i] == 1) && (predict[i] == 0)) fn++;
101  }
102 
103  // after calculate value we can build confusion matrix
104  Matrix CM(2,2);
105  CM << tn, fn,
106  fp, tp;
107  return CM;
108  }

125 
126  Scalar accuracy = CM.diagonal().sum() / CM.sum();
127  Scalar precision = ((CM.diagonal().array()) / (CM.rowwise().sum().array())).mean();
128  Scalar recall = ((CM.diagonal().array().transpose()) / (CM.colwise().sum().array())).mean();
129  Scalar f1_score = 2 * (precision * recall) / (precision + recall);
130 
131  std::cout << "accuracy = " << accuracy << std::endl;
