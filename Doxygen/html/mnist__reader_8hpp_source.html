80 bool read_mnist_image_file_flat(Container& images, const std::string& path, std::size_t limit, std::size_t start = 0) {
81  auto buffer = read_mnist_file(path, 0x803);
82 
83  if (buffer) {
84  auto count = read_header(buffer, 1);
85  auto rows = read_header(buffer, 2);
86  auto columns = read_header(buffer, 3);
87 
88  //Skip the header
89  //Cast to unsigned char is necessary cause signedness of char is
90  //platform-specific
91  auto image_buffer = reinterpret_cast<unsigned char*>(buffer.get() + 16);
92 
93  if (limit > 0 && count > limit) {
94  count = static_cast<unsigned int>(limit);
95  }
96 
97  // Ignore "start" first elements
98  image_buffer += start * (rows * columns);
99 
100  for (size_t i = 0; i < count; ++i) {
101  for (size_t j = 0; j < rows * columns; ++j) {
102  images(i)[j] = *image_buffer++;
103  }
104  }
105 
106  return true;
107  } else {
108  return false;
109  }
110 }

119 template <template <typename...> class Container = std::vector, typename Image, typename Functor>
120 void read_mnist_image_file(Container<Image>& images, const std::string& path, std::size_t limit, Functor func) {
121  auto buffer = read_mnist_file(path, 0x803);
122 
123  if (buffer) {
124  auto count = read_header(buffer, 1);
125  auto rows = read_header(buffer, 2);
126  auto columns = read_header(buffer, 3);
127 
128  //Skip the header
129  //Cast to unsigned char is necessary cause signedness of char is
130  //platform-specific
131  auto image_buffer = reinterpret_cast<unsigned char*>(buffer.get() + 16);
132 
133  if (limit > 0 && count > limit) {
134  count = static_cast<unsigned int>(limit);
135  }
136 
137  images.reserve(count);
138 
139  for (size_t i = 0; i < count; ++i) {
140  images.push_back(func());
141 
142  for (size_t j = 0; j < rows * columns; ++j) {
143  auto pixel = *image_buffer++;
144  images[i][j] = static_cast<typename Image::value_type>(pixel);
145  }
146  }
147  }
