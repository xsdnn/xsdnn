99  {
100  const unsigned long nlayer = count_layers();
101 
102  if (nlayer <= 0) { return; }
103 
104  // Создадим указатель на первый и последний (скрытый, но не выходной)
105  // слой сетки, это поможет в дальнейшем
106 
107  Layer* first_layer = m_layers[0];
108  Layer* last_layer = m_layers[nlayer - 1];
109 
110  // Начнем распространение с конца сетки
111  m_output->check_target_data(target);
112  m_output->evaluate(last_layer->output(), target);
113 
114  // Если скрытый слой всего один, то 'prev_layer_data' будут выходными данными
115 
116  if (nlayer == 1)
117  {
118  first_layer->backprop(input, last_layer->backprop_data());
119  return;
120  }
121 
122  // Если это условие не выполнено, то вычисляем градиент для последнего скрытого слоя
123  last_layer->backprop(m_layers[nlayer - 2]->output(), m_output->backprop_data());
124 
125  // Теперь пробегаемся по всем слоям и вычисляем градиенты
126 
127  for (unsigned long i = nlayer - 2; i > 0; --i)
128  {
129  m_layers[i]->backprop(m_layers[i - 1]->output(),
130  m_layers[i + 1]->backprop_data());
131  }
132 
133  // Теперь вычисляем грады для нулевого - входного слоя сетки
134 
135  first_layer->backprop(input, m_layers[1]->backprop_data());
136 
137  // На этом backprop окончен
138  }
